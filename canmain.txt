#include<lpc21xx.h>
#include "can2_driver.c"
#include "header.h"
#include "uart0_driver.c"
#include "canlcd.c"

#define sw1 14
#define sw2 15
#define sw3 16
  int flag1=0;
  int flag2 =0;
  int flag3 =0;
int main(){	

	CAN2_MSG m1, m2, m3;

	IODIR0 = sw1|sw2|sw3;
	LCD_init();

	can2_init();

	uart0_init( 115200);

	/*sending data frame*/

	m1.id=0x000001AF;

	m1.rtr=0;//data frame

	m1.dlc=4;

	m1.byteA=0xDDCCBBAA;

	m1.byteB=0;

	

	m2.id=0x000001BF;

	m2.rtr=0;//data frame

	m2.dlc=4;

	m2.byteA=0xDDCCBBAA;

	m2.byteB=0;
	
	

	m3.id=0x000001CF;

	m3.rtr=0;//data frame

	m3.dlc=4;

	m3.byteA=0xDDCCBBAA;

	m3.byteB=0;
    str (" transfering");
    uart0_tx_string("TRANSFERING\r\n");

	while(1)
		{
			 
		if((IOPIN0>> sw1 &1)== 0){

		uart0_tx_string("MESSAGE ID:");
		uart0_tx_hex( 0x000001AF);
		uart0_tx_string("\r\n");
		  if(flag1 == 0){
		  flag1 =1;
				  LCD_command(0x01);
				   LCD_command(0x80);
	     str("LEFTINDICATORON");
		 uart0_tx_string("LEFT INDICATOR ON\r\n");
		 	
		can2_tx(m1);//data-frame

		delay_milli(200);


		}  else{
		        flag1 =0;
		       	 LCD_command(0x01);
				   LCD_command(0x80);
		 str("LEFTINDICATOROFF");
		 uart0_tx_string("LEFT INDICATOR OFF\r\n");
		      
		    
			can2_tx(m1);//data-frame

			delay_milli(200);
		 }
		} 
		
	  else if((IOPIN0>> sw2 &1)== 0){

		uart0_tx_string("MESSAGE ID:");
		uart0_tx_hex( 0x000001BF);
		uart0_tx_string("\r\n");

		if(flag2 == 0){
		
		  					flag2 =1;
		 LCD_command(0x01);
		 LCD_command(0x80);
		str("RIGHTINDICATORON");
		uart0_tx_string("RIGHT INDICATOR ON\r\n");

		can2_tx(m2);//data-frame
		delay_milli(200);

		  }else{
		  					flag2 =0;

				 str("RIGHTINDICATOROFF");
	        	uart0_tx_string("RIGHT INDICATOR OFF\r\n");
				
	        	can2_tx(m2);//data-frame
				delay_milli(200);
		  }

		}
		else if((IOPIN0>> sw3 &1)== 0){
		uart0_tx_hex( 0x000001CF);
		uart0_tx_string("\r\n");

		   if(flag3 == 0){

		 flag3 =1;
		 LCD_command(0x01);
		 LCD_command(0x83);

   		str("WIPER ON");	
		uart0_tx_string("WIPER ON\r\n");
		can2_tx(m3);//data-frame
		delay_milli(200);
		
		}else{
				 flag3 = 0;
				  LCD_command(0x01);
		 LCD_command(0x83);
		  str("WIPER OFF");	

		uart0_tx_string("WIPER OFF\r\n");

		can2_tx(m3);//data-frame
		delay_milli(200);
		}
		}
		}
   
}
